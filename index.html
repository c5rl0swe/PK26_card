<script>

const canvas = document.getElementById('canvas');

const ctx = canvas.getContext('2d');


let userImage = null;

let logo = new Image();

logo.src = "logo.png";


document.getElementById('imageUpload').addEventListener('change', e => {

    const reader = new FileReader();

    reader.onload = evt => {

        const img = new Image();

        img.onload = () => {

            userImage = img;

            draw();

        }

        img.src = evt.target.result;

    }

    reader.readAsDataURL(e.target.files[0]);

});


document.getElementById('nameInput').addEventListener('input', draw);

document.getElementById('jobInput').addEventListener('input', draw);


logo.onload = draw;


function draw() {

    // BG

    ctx.fillStyle = "#3a3a3a";

    ctx.fillRect(0, 0, canvas.width, canvas.height);


    // LOGO

    if (logo.complete) {

        ctx.drawImage(logo, 60, 60, 260, 120);

    }


    // DIVIDER

    ctx.strokeStyle = "#555";

    ctx.beginPath();

    ctx.moveTo(60, 220);

    ctx.lineTo(1020, 220);

    ctx.stroke();


    // IMAGE CIRCLE

    const cx = 300;

    const cy = 520;

    const r = 180;


    if (userImage) {

        ctx.save();

        ctx.beginPath();

        ctx.arc(cx, cy, r, 0, Math.PI * 2);

        ctx.closePath();

        ctx.clip();


        const scale = Math.min(

            (r * 2) / userImage.width,

            (r * 2) / userImage.height

        );


        const w = userImage.width * scale;

        const h = userImage.height * scale;


        ctx.drawImage(

            userImage,

            cx - w / 2,

            cy - h / 2,

            w,

            h

        );


        ctx.restore();

    }


    // CIRCLE BORDER

    ctx.strokeStyle = "#FFC400";

    ctx.lineWidth = 12;

    ctx.beginPath();

    ctx.arc(cx, cy, r + 6, 0, Math.PI * 2);

    ctx.stroke();


    // TEXT

    const name = document.getElementById('nameInput').value || "Your Name";

    const job = document.getElementById('jobInput').value || "Job Title";


    ctx.fillStyle = "#fff";

    ctx.font = "700 56px Hanken Grotesk";

    ctx.fillText(name, 520, 500);


    // underline

    ctx.strokeStyle = "#FFC400";

    ctx.lineWidth = 4;

    ctx.beginPath();

    ctx.moveTo(520, 520);

    ctx.lineTo(820, 520);

    ctx.stroke();


    ctx.fillStyle = "#ccc";

    ctx.font = "400 34px Hanken Grotesk";

    ctx.fillText(job, 520, 580);


    // BOTTOM LEFT

    ctx.fillStyle = "#fff";

    ctx.font = "700 36px Hanken Grotesk";

    ctx.fillText("ALL IN!! LET'S GO!", 60, 900);


    ctx.strokeStyle = "#FFC400";

    ctx.lineWidth = 4;

    ctx.beginPath();

    ctx.moveTo(60, 915);

    ctx.lineTo(380, 915);

    ctx.stroke();


    ctx.fillStyle = "#aaa";

    ctx.font = "28px Hanken Grotesk";

    ctx.fillText("May 4th – 8th", 60, 960);


    // RIGHT

    ctx.fillStyle = "#ccc";

    ctx.font = "700 28px Hanken Grotesk";

    ctx.fillText("PRESIDENT'S", 720, 820);


    ctx.fillStyle = "#fff";

    ctx.font = "700 80px Hanken Grotesk";

    ctx.fillText("KAIZEN", 720, 900);

}


function download() {

    const link = document.createElement('a');

    link.download = "kaizen.png";

    link.href = canvas.toDataURL("image/png");

    link.click();

}


draw();

</script>
